import serial
import time

# Establish a connection to the Arduino
ser = serial.Serial('COM3', 9600)  # Replace 'COM3' with your actual COM port
time.sleep(2)  # Wait for the connection to establish

def send_command(movement, motor_duration, led_duration):
    """
    Send command to the Arduino to control the motor and LED.

    Args:
    movement (str): 'FORWARD' or 'BACKWARD'
    motor_duration (int): Duration in seconds for motor action
    led_duration (int): Duration in seconds for the LED to be on
    """
    command = f"{movement} {motor_duration} LED {led_duration}"
    ser.write((command + '\n').encode())
    print(f"Command sent: {command}")  # Print the command for debugging

# Example usage
send_command("FORWARD", 5, 3)  # Move forward for 5 seconds and LED on for 3 seconds
time.sleep(6)  # Wait a bit longer than the longest duration (motor movement)

send_command("BACKWARD", 3, 2)  # Move backward for 3 seconds and LED on for 2 seconds
time.sleep(4)  # Wait a bit longer than the longest duration (motor movement)

ser.close()  # Close the serial connection
